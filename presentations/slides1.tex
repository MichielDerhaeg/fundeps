\begin{frame}[fragile]
    \frametitle{Functional Dependencies}
    \begin{lstlisting}
class Collection c e | c -> e where
    singleton :: e -> c

instance Collection [a] a where
    singleton x = [x]

singleton2 :: (Coll c1 e, Coll c2 c1 ) => e -> c2
singleton2 c = singleton (singleton c)
    \end{lstlisting}
