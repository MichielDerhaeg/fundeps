    \textbf{The Superclass Issue}
    \begin{lstlisting}
class Eq a where
    (==) :: a -> a -> Bool

class Eq a => Ord a where
    (<=) :: a -> a -> Bool

(<) :: (Eq a, Ord a) => a -> a -> Bool
(<) a b | a == b = False
        | a <= b = True
        | otherwise = False
